  $(function() {
    var $mountedElem  = $('.download-archive'),
        $archiveModal = $('.project-archive-modal'),
        $archiveName  = $archiveModal.find('.header .name'),
        $downloadBtn  = $archiveModal.find('.download-btn'),
        $captchaInput = $archiveModal.find('#captcha'),
        $captchaImg   = $archiveModal.find('.captcha_img'),
        $message      = $archiveModal.find('.form-message'),
        isLogin       = false,
        loginUrl      = '/login',
        archiveUrl    = null,
        ref           = null,
        sha           = null,
        captcha       = null,

    // 下载文件

      captcha = $captchaInput.val()
      downloadArchive()
      return false
    })
  
    function downloadArchive() {
      $.ajax({
        type: 'POST',
        url: '/DSN2002/ULRanalysis/repository/download',
        data: { sha: sha, format: format, captcha: captcha },
        beforeSend: function() {
          $downloadBtn.addClass('loading')
          $message.removeClass('info error').hide()
        },
        success: function(data) {
          $downloadBtn.removeClass('loading')
          if (data.code == 200) {
            location.href = archiveUrl + '&captcha=' + captcha
            $archiveModal.modal('hide')
          } else if (data.code == 201) {
            delayTimers['checkArchive'] = setInterval(checkArchive, 3000)
            $downloadBtn.addClass('disabled')
            $message.addClass('info').text(data.msg).show()
            $captchaImg.trigger('click')
          } else if (data.code == 401) {
            $message.addClass('error').text(data.msg).show()
          }
        }
      })
    }
  
    function checkArchive() {
      $.get('/DSN2002/ULRanalysis/repository/check_archive', { sha: sha, format: format }, function(data) {
        if (data.code == 200) {
          location.href = archiveUrl + '&captcha=' + captcha
          clearInterval(delayTimers['checkArchive'])
          $archiveModal.modal('hide')
        } else if (data.code == 401) {
          $message.removeClass('info').addClass('error').text(data.msg)
          clearInterval(delayTimers['checkArchive'])
        }
      })
    }
  
    // 检验验证码
    $captchaInput.on('keyup', function() {
      delayRun('captcha', checkCaptcha)
    })
  
    $captchaImg.on('click', function() {
      $captchaInput.val('')
      $captchaInput.closest('.field').removeClass('error')
      $downloadBtn.addClass('disabled')
    })
  
    function checkCaptcha() {
      var $field  = $captchaInput.closest('.field')
          captcha = $captchaInput.val()
  
      if (captcha.length < 6) {
        $downloadBtn.addClass('disabled')
        $field.removeClass('error')
        return
      }
  
      $.ajax({
        type: 'POST',
        url: '/check',
        data: { do: 'captcha', val: captcha },
        success: function(data) {
          if (data == 1) {
            $field.removeClass('error')
            $downloadBtn.removeClass('disabled')
          } else {
            $field.addClass('error')
            $field.find('.message').text(data)
            $downloadBtn.addClass('disabled')
          }
        }
      })
    }
  
    function stopDelayRun(task) {
      if (delayTimers[task]) {
        clearTimeout(delayTimers[task])
        delayTimers[task] = null
      }
    }
  
    function delayRun(task, callback) {
      stopDelayRun(task)
      delayTimers[task] = setTimeout(callback, 1000)
    }
  })
